49 static void internalMessageFallback(KMessage::MessageType messageType, const QString &text, const QString &caption)
50 {
51  QString prefix;
52  switch (messageType) {
53  case KMessage::Error:
54  prefix = QStringLiteral("ERROR: ");
55  break;
56  case KMessage::Fatal:
57  prefix = QStringLiteral("FATAL: ");
58  break;
59  case KMessage::Information:
60  prefix = QStringLiteral("INFORMATION: ");
61  break;
62  case KMessage::Sorry:
63  prefix = QStringLiteral("SORRY: ");
64  break;
65  case KMessage::Warning:
66  prefix = QStringLiteral("WARNING: ");
67  break;
68  }
69 
70  QString message;
71 
72  if (!caption.isEmpty()) {
73  message += QLatin1Char('(') + caption + QLatin1Char(')');
74  }
75 
76  message += prefix + text;
77 
78  // Show a message to the developer to setup a KMessageHandler
79  std::cerr << "WARNING: Please setup an KMessageHandler with KMessage::setMessageHandler to display message propertly." << std::endl;
80  // Show message to stdout
81  std::cerr << qPrintable(message) << std::endl;
82 }

90 void KMessage::message(KMessage::MessageType messageType, const QString &text, const QString &caption)
91 {
92  // Use current message handler if available, else use stdout
93  if (s_messageHandler()->handler()) {

95  } else {
96  internalMessageFallback(messageType, text, caption);
