53 int DesktopToJson::runMain()
54 {
55  if (!m_parser->isSet(input)) {
56  m_parser->showHelp(1);
57  return 1;
58  }
59  if (m_parser->isSet(verbose)) {
60  DesktopFileParser::s_verbose = true;
61  }
62  if (m_parser->isSet(compat)) {
63  DesktopFileParser::s_compatibilityMode = true;
64  }
65  if (!resolveFiles()) {
66  qCCritical(DESKTOPPARSER) << "Failed to resolve filenames" << m_inFile << m_outFile << endl;
67  return 1;
68  }
69 
70 #pragma message("TODO: make it an error if one of the service type files is invalid or not found")
71  QStringList serviceTypes = m_parser->values(serviceTypesOption);
72  return convert(m_inFile, m_outFile, serviceTypes) ? EXIT_SUCCESS : EXIT_FAILURE;
73 }

75 bool DesktopToJson::resolveFiles()
76 {
77  if (m_parser->isSet(input)) {
78  m_inFile = m_parser->value(input);
79  const QFileInfo fi(m_inFile);
80  if (!fi.exists()) {
81  qCCritical(DESKTOPPARSER) << "File not found: " << m_inFile << endl;
82  return false;
83  }
84  if (!fi.isAbsolute()) {
85  m_inFile = fi.absoluteFilePath();
86  }
87  }
88 
89  if (m_parser->isSet(output)) {
90  m_outFile = m_parser->value(output);
91  } else if (!m_inFile.isEmpty()) {
92  m_outFile = m_inFile;
93  m_outFile.replace(QStringLiteral(".desktop"), QStringLiteral(".json"));
94  }
95 
96  return m_inFile != m_outFile && !m_inFile.isEmpty() && !m_outFile.isEmpty();
97 }

99 void DesktopFileParser::convertToCompatibilityJson(const QString &key, const QString &value, QJsonObject &json, int lineNr)
100 {
101  // XXX: Hidden=true doesn't make sense with json plugins since the metadata is inside the .so
102  static const QStringList boolkeys = QStringList()
103  << QStringLiteral("Hidden") << QStringLiteral("X-KDE-PluginInfo-EnabledByDefault");
104  static const QStringList stringlistkeys = QStringList()
105  << QStringLiteral("X-KDE-ServiceTypes") << QStringLiteral("X-KDE-PluginInfo-Depends");
106  if (boolkeys.contains(key)) {
107  // should only be lower case, but be tolerant here
108  if (value.toLower() == QLatin1String("true")) {
109  json[key] = true;
110  } else {
111  if (value.toLower() != QLatin1String("false")) {
112  qCWarning(DESKTOPPARSER).nospace() << "Expected boolean value for key \"" << key
113  << "\" at line " << lineNr << "but got \"" << value << "\" instead.";
114  }
115  json[key] = false;
116  }
117  } else if (stringlistkeys.contains(key)) {
118  json[key] = QJsonArray::fromStringList(DesktopFileParser::deserializeList(value));
119  } else {
120  json[key] = value;
121  }
122 }

124 bool DesktopToJson::convert(const QString &src, const QString &dest, const QStringList& serviceTypes)
125 {
126 
127  QJsonObject json;
128  DesktopFileParser::convert(src, serviceTypes, json, nullptr);
129 
130  if (DesktopFileParser::s_compatibilityMode) {
131  Q_ASSERT(json.value(QStringLiteral("KPlugin")).toObject().isEmpty());
132  json.remove(QStringLiteral("KPlugin"));
133  }
134  QJsonDocument jdoc;
135  jdoc.setObject(json);
136 
137  QFile file(dest);
138  if (!file.open(QIODevice::WriteOnly | QIODevice::Text)) {
139  qCCritical(DESKTOPPARSER) << "Failed to open " << dest << endl;
140  return false;
141  }
142 
143  file.write(jdoc.toJson());
144  qCDebug(DESKTOPPARSER) << "Generated " << dest << endl;
145  return true;
146 }
