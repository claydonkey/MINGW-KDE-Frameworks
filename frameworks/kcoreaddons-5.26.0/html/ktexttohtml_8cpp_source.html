80 QString KTextToHTMLHelper::getEmailAddress()
81 {
82  QString address;
83 
84  if (mText[mPos] == QLatin1Char('@')) {
85  // the following characters are allowed in a dot-atom (RFC 2822):
86  // a-z A-Z 0-9 . ! # $ % & ' * + - / = ? ^ _ ` { | } ~
87  static const QString allowedSpecialChars = QStringLiteral(".!#$%&'*+-/=?^_`{|}~");
88 
89  // determine the local part of the email address
90  int start = mPos - 1;
91  while (start >= 0 && mText[start].unicode() < 128 &&
92  (mText[start].isLetterOrNumber() ||
93  mText[start] == QLatin1Char('@') || // allow @ to find invalid email addresses
94  allowedSpecialChars.indexOf(mText[start]) != -1)) {
95  if (mText[start] == QLatin1Char('@')) {
96  return QString(); // local part contains '@' -> no email address
97  }
98  --start;
99  }
100  ++start;
101  // we assume that an email address starts with a letter or a digit
102  while ((start < mPos) && !mText[start].isLetterOrNumber()) {
103  ++start;
104  }
105  if (start == mPos) {
106  return QString(); // local part is empty -> no email address
107  }
108 
109  // determine the domain part of the email address
110  int dotPos = INT_MAX;
111  int end = mPos + 1;
112  while (end < mText.length() &&
113  (mText[end].isLetterOrNumber() ||
114  mText[end] == QLatin1Char('@') || // allow @ to find invalid email addresses
115  mText[end] == QLatin1Char('.') ||
116  mText[end] == QLatin1Char('-'))) {
117  if (mText[end] == QLatin1Char('@')) {
118  return QString(); // domain part contains '@' -> no email address
119  }
120  if (mText[end] == QLatin1Char('.')) {
121  dotPos = qMin(dotPos, end); // remember index of first dot in domain
122  }
123  ++end;
124  }
125  // we assume that an email address ends with a letter or a digit
126  while ((end > mPos) && !mText[end - 1].isLetterOrNumber()) {
127  --end;
128  }
129  if (end == mPos) {
130  return QString(); // domain part is empty -> no email address
131  }
132  if (dotPos >= end) {
133  return QString(); // domain part doesn't contain a dot
134  }
135 
136  if (end - start > mMaxAddressLen) {
137  return QString(); // too long -> most likely no email address
138  }
139  address = mText.mid(start, end - start);
140 
141  mPos = end - 1;
142  }
143  return address;
144 }

146 bool KTextToHTMLHelper::atUrl()
147 {
148  // the following characters are allowed in a dot-atom (RFC 2822):
149  // a-z A-Z 0-9 . ! # $ % & ' * + - / = ? ^ _ ` { | } ~
150  static const QString allowedSpecialChars = QStringLiteral(".!#$%&'*+-/=?^_`{|}~");
151 
152  // the character directly before the URL must not be a letter, a number or
153  // any other character allowed in a dot-atom (RFC 2822).
154  if ((mPos > 0) &&
155  (mText[mPos - 1].isLetterOrNumber() ||
156  (allowedSpecialChars.indexOf(mText[mPos - 1]) != -1))) {
157  return false;
158  }
159 
160  QChar ch = mText[mPos];
161  return
162  (ch == QLatin1Char('h') && (mText.mid(mPos, 7) == QLatin1String("http://") ||
163  mText.mid(mPos, 8) == QLatin1String("https://"))) ||
164  (ch == QLatin1Char('v') && mText.mid(mPos, 6) == QLatin1String("vnc://")) ||
165  (ch == QLatin1Char('f') && (mText.mid(mPos, 7) == QLatin1String("fish://") ||
166  mText.mid(mPos, 6) == QLatin1String("ftp://") ||
167  mText.mid(mPos, 7) == QLatin1String("ftps://"))) ||
168  (ch == QLatin1Char('s') && (mText.mid(mPos, 7) == QLatin1String("sftp://") ||
169  mText.mid(mPos, 6) == QLatin1String("smb://"))) ||
170  (ch == QLatin1Char('m') && mText.mid(mPos, 7) == QLatin1String("mailto:")) ||
171  (ch == QLatin1Char('w') && mText.mid(mPos, 4) == QLatin1String("www.")) ||
172  (ch == QLatin1Char('f') && (mText.mid(mPos, 4) == QLatin1String("ftp.") ||
173  mText.mid(mPos, 7) == QLatin1String("file://"))) ||
174  (ch == QLatin1Char('n') && mText.mid(mPos, 5) == QLatin1String("news:"));
175 }

195 QString KTextToHTMLHelper::getUrl()
196 {
197  QString url;
198  if (atUrl()) {
199  // NOTE: see http://tools.ietf.org/html/rfc3986#appendix-A and especially appendix-C
200  // Appendix-C mainly says, that when extracting URLs from plain text, line breaks shall
201  // be allowed and should be ignored when the URI is extracted.
202 
203  // This implementation follows this recommendation and
204  // allows the URL to be enclosed within different kind of brackets/quotes
205  // If an URL is enclosed, whitespace characters are allowed and removed, otherwise
206  // the URL ends with the first whitespace
207  // Also, if the URL is enclosed in brackets, the URL itself is not allowed
208  // to contain the closing bracket, as this would be detected as the end of the URL
209 
210  QChar beforeUrl, afterUrl;
211 
212  // detect if the url has been surrounded by brackets or quotes
213  if (mPos > 0) {
214  beforeUrl = mText[mPos - 1];
215 
216  /*if ( beforeUrl == '(' ) {
217  afterUrl = ')';
218  } else */if (beforeUrl == QLatin1Char('[')) {
219  afterUrl = QLatin1Char(']');
220  } else if (beforeUrl == QLatin1Char('<')) {
221  afterUrl = QLatin1Char('>');
222  } else if (beforeUrl == QLatin1Char('>')) { // for e.g. <link>http://.....</link>
223  afterUrl = QLatin1Char('<');
224  } else if (beforeUrl == QLatin1Char('"')) {
225  afterUrl = QLatin1Char('"');
226  }
227  }
228 
229  url.reserve(mMaxUrlLen); // avoid allocs
230  int start = mPos;
231  while ((mPos < mText.length()) &&
232  (mText[mPos].isPrint() || mText[mPos].isSpace()) &&
233  ((afterUrl.isNull() && !mText[mPos].isSpace()) ||
234  (!afterUrl.isNull() && mText[mPos] != afterUrl))) {
235  if (!mText[mPos].isSpace()) { // skip whitespace
236  url.append(mText[mPos]);
237  if (url.length() > mMaxUrlLen) {
238  break;
239  }
240  }
241 
242  ++mPos;
243  }
244 
245  if (isEmptyUrl(url) || (url.length() > mMaxUrlLen)) {
246  mPos = start;
247  url.clear();
248  } else {
249  --mPos;
250  }
251  }
252 
253  // HACK: This is actually against the RFC. However, most people don't properly escape the URL in
254  // their text with "" or <>. That leads to people writing an url, followed immediatley by
255  // a dot to finish the sentence. That would lead the parser to include the dot in the url,
256  // even though that is not wanted. So work around that here.
257  // Most real-life URLs hopefully don't end with dots or commas.
258  QList<QChar> wordBoundaries;
259  wordBoundaries << QLatin1Char('.') << QLatin1Char(',') << QLatin1Char(':') << QLatin1Char('!') << QLatin1Char('?') << QLatin1Char(')') << QLatin1Char('>');
260  if (url.length() > 1) {
261  do {
262  if (wordBoundaries.contains(url.at(url.length() - 1))) {
263  url.chop(1);
264  --mPos;
265  } else {
266  break;
267  }
268  } while (url.length() > 1);
269  }
270 
271  return url;
272 }

274 QString KTextToHTMLHelper::highlightedText()
275 {
276  // formating symbols must be prepended with a whitespace
277  if ((mPos > 0) && !mText[mPos - 1].isSpace()) {
278  return QString();
279  }
280 
281  const QChar ch = mText[mPos];
282  if (ch != QLatin1Char('/') && ch != QLatin1Char('*') && ch != QLatin1Char('_') && ch != QLatin1Char('-')) {
283  return QString();
284  }
285 
286  QRegExp re =
287  QRegExp(QStringLiteral("\\%1((\\w+)([\\s-']\\w+)*( ?[,.:\\?!;])?)\\%2").arg(ch).arg(ch));
288  re.setMinimal(true);
289  if (re.indexIn(mText, mPos) == mPos) {
290  int length = re.matchedLength();
291  // there must be a whitespace after the closing formating symbol
292  if (mPos + length < mText.length() && !mText[mPos + length].isSpace()) {
293  return QString();
294  }
295  mPos += length - 1;
296  switch (ch.toLatin1()) {
297  case '*':
298  return QLatin1String("<b>*") + re.cap(1) + QLatin1String("*</b>");
299  case '_':
300  return QLatin1String("<u>_") + re.cap(1) + QLatin1String("_</u>");
301  case '/':
302  return QLatin1String("<i>/") + re.cap(1) + QLatin1String("/</i>");
303  case '-':
304  return QLatin1String("<strike>-") + re.cap(1) + QLatin1String("-</strike>");
305  }
306  }
307  return QString();
308 }

328 QString KTextToHTML::convertToHtml(const QString &plainText, const KTextToHTML::Options &flags, int maxUrlLen, int maxAddressLen)
329 {
330  KTextToHTMLHelper helper(plainText, maxUrlLen, maxAddressLen);
331 
332  QString str;
333  QString result(static_cast<QChar *>(Q_NULLPTR), helper.mText.length() * 2);
334  QChar ch;
335  int x;
336  bool startOfLine = true;
337 
338  for (helper.mPos = 0, x = 0; helper.mPos < helper.mText.length();
339  ++helper.mPos, ++x) {
340  ch = helper.mText[helper.mPos];
341  if (flags & PreserveSpaces) {
342  if (ch == QLatin1Char(' ')) {
343  if (helper.mPos + 1 < helper.mText.length()) {
344  if (helper.mText[helper.mPos + 1] != QLatin1Char(' ')) {
345 
346  // A single space, make it breaking if not at the start or end of the line
347  const bool endOfLine = helper.mText[helper.mPos + 1] == QLatin1Char('\n');
348  if (!startOfLine && !endOfLine) {
349  result += QLatin1Char(' ');
350  } else {
351  result += QLatin1String("&nbsp;");
352  }
353  } else {
354 
355  // Whitespace of more than one space, make it all non-breaking
356  while (helper.mPos < helper.mText.length() && helper.mText[helper.mPos] == QLatin1Char(' ')) {
357  result += QLatin1String("&nbsp;");
358  ++helper.mPos;
359  ++x;
360  }
361 
362  // We incremented once to often, undo that
363  --helper.mPos;
364  --x;
365  }
366  } else {
367  // Last space in the text, it is non-breaking
368  result += QLatin1String("&nbsp;");
369  }
370 
371  if (startOfLine) {
372  startOfLine = false;
373  }
374  continue;
375  } else if (ch == QLatin1Char('\t')) {
376  do {
377  result += QLatin1String("&nbsp;");
378  ++x;
379  } while ((x & 7) != 0);
380  --x;
381  startOfLine = false;
382  continue;
383  }
384  }
385  if (ch == QLatin1Char('\n')) {
386  result += QLatin1String("<br />\n"); // Keep the \n, so apps can figure out the quoting levels correctly.
387  startOfLine = true;
388  x = -1;
389  continue;
390  }
391 
392  startOfLine = false;
393  if (ch == QLatin1Char('&')) {
394  result += QLatin1String("&amp;");
395  } else if (ch == QLatin1Char('"')) {
396  result += QLatin1String("&quot;");
397  } else if (ch == QLatin1Char('<')) {
398  result += QLatin1String("&lt;");
399  } else if (ch == QLatin1Char('>')) {
400  result += QLatin1String("&gt;");
401  } else {
402  const int start = helper.mPos;
403  if (!(flags & IgnoreUrls)) {
404  str = helper.getUrl();
405  if (!str.isEmpty()) {
406  QString hyperlink;
407  if (str.left(4) == QLatin1String("www.")) {
408  hyperlink = QLatin1String("http://") + str;
409  } else if (str.left(4) == QLatin1String("ftp.")) {
410  hyperlink = QLatin1String("ftp://") + str;
411  } else {
412  hyperlink = str;
413  }
414 
415  result += QLatin1String("<a href=\"") + hyperlink + QLatin1String("\">") + str.toHtmlEscaped() + QLatin1String("</a>");
416  x += helper.mPos - start;
417  continue;
418  }
419  str = helper.getEmailAddress();
420  if (!str.isEmpty()) {
421  // len is the length of the local part
422  int len = str.indexOf(QLatin1Char('@'));
423  QString localPart = str.left(len);
424 
425  // remove the local part from the result (as '&'s have been expanded to
426  // &amp; we have to take care of the 4 additional characters per '&')
427  result.truncate(result.length() -
428  len - (localPart.count(QLatin1Char('&')) * 4));
429  x -= len;
430 
431  result += QLatin1String("<a href=\"mailto:") + str + QLatin1String("\">") + str + QLatin1String("</a>");
432  x += str.length() - 1;
433  continue;
434  }
435  }
436  if (flags & HighlightText) {
437  str = helper.highlightedText();
438  if (!str.isEmpty()) {
439  result += str;
440  x += helper.mPos - start;
441  continue;
442  }
443  }
444  result += ch;
445  }
446  }
447 
448  if (flags & ReplaceSmileys) {
449  QStringList exclude;
450  exclude << QStringLiteral("(c)") << QStringLiteral("(C)") << QStringLiteral("&gt;:-(") << QStringLiteral("&gt;:(") << QStringLiteral("(B)") << QStringLiteral("(b)") << QStringLiteral("(P)") << QStringLiteral("(p)");
451  exclude << QStringLiteral("(O)") << QStringLiteral("(o)") << QStringLiteral("(D)") << QStringLiteral("(d)") << QStringLiteral("(E)") << QStringLiteral("(e)") << QStringLiteral("(K)") << QStringLiteral("(k)");
452  exclude << QStringLiteral("(I)") << QStringLiteral("(i)") << QStringLiteral("(L)") << QStringLiteral("(l)") << QStringLiteral("(8)") << QStringLiteral("(T)") << QStringLiteral("(t)") << QStringLiteral("(G)");
453  exclude << QStringLiteral("(g)") << QStringLiteral("(F)") << QStringLiteral("(f)") << QStringLiteral("(H)");
454  exclude << QStringLiteral("8)") << QStringLiteral("(N)") << QStringLiteral("(n)") << QStringLiteral("(Y)") << QStringLiteral("(y)") << QStringLiteral("(U)") << QStringLiteral("(u)") << QStringLiteral("(W)") << QStringLiteral("(w)");
455 
456  result = helper.emoticonsInterface()->parseEmoticons(result, true, exclude);
457  }
458 
459  return result;
460 }
