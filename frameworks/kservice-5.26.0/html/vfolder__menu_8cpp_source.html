142 VFolderMenu::takeSubMenu(SubMenu *parentMenu, const QString &menuName)
143 {
144  const int i = menuName.indexOf('/');
145  const QString s1 = i > 0 ? menuName.left(i) : menuName;
146  const QString s2 = menuName.mid(i + 1);
147 
148  // Look up menu
149  for (QList<SubMenu *>::Iterator it = parentMenu->subMenus.begin(); it != parentMenu->subMenus.end(); ++it) {
150  SubMenu *menu = *it;

152  if (i == -1) {
153  // Take it out
154  parentMenu->subMenus.erase(it);
155  return menu;
156  } else {
157  return takeSubMenu(menu, s2);
158  }
159  }
160  }
161  return 0; // Not found

164 void
165 VFolderMenu::mergeMenu(SubMenu *menu1, SubMenu *menu2, bool reversePriority)
166 {
167  if (m_track) {
168  track(m_trackId, menu1->name, menu1->items, menu1->excludeItems, menu2->items, QStringLiteral("Before MenuMerge w. %1 (incl)").arg(menu2->name));
169  track(m_trackId, menu1->name, menu1->items, menu1->excludeItems, menu2->excludeItems, QStringLiteral("Before MenuMerge w. %1 (excl)").arg(menu2->name));
170  }
171  if (reversePriority) {
172  // Merge menu1 with menu2, menu1 takes precedent
173  excludeItems(menu2->items, menu1->excludeItems);
174  includeItems(menu1->items, menu2->items);
175  excludeItems(menu2->excludeItems, menu1->items);
176  includeItems(menu1->excludeItems, menu2->excludeItems);
177  } else {
178  // Merge menu1 with menu2, menu2 takes precedent
179  excludeItems(menu1->items, menu2->excludeItems);
180  includeItems(menu1->items, menu2->items);
181  includeItems(menu1->excludeItems, menu2->excludeItems);
182  menu1->isDeleted = menu2->isDeleted;
183  }
184  while (!menu2->subMenus.isEmpty()) {
185  SubMenu *subMenu = menu2->subMenus.takeFirst();
186  insertSubMenu(menu1, subMenu->name, subMenu, reversePriority);
187  }
188 
189  if (reversePriority) {
190  // Merge menu1 with menu2, menu1 takes precedent
191  if (menu1->directoryFile.isEmpty()) {
192  menu1->directoryFile = menu2->directoryFile;
193  }
194  if (menu1->defaultLayoutNode.isNull()) {
195  menu1->defaultLayoutNode = menu2->defaultLayoutNode;
196  }
197  if (menu1->layoutNode.isNull()) {
198  menu1->layoutNode = menu2->layoutNode;
199  }
200  } else {
201  // Merge menu1 with menu2, menu2 takes precedent
202  if (!menu2->directoryFile.isEmpty()) {
203  menu1->directoryFile = menu2->directoryFile;
204  }
205  if (!menu2->defaultLayoutNode.isNull()) {
206  menu1->defaultLayoutNode = menu2->defaultLayoutNode;
207  }
208  if (!menu2->layoutNode.isNull()) {
209  menu1->layoutNode = menu2->layoutNode;
210  }
211  }
212 
213  if (m_track) {
214  track(m_trackId, menu1->name, menu1->items, menu1->excludeItems, menu2->items, QStringLiteral("After MenuMerge w. %1 (incl)").arg(menu2->name));
215  track(m_trackId, menu1->name, menu1->items, menu1->excludeItems, menu2->excludeItems, QStringLiteral("After MenuMerge w. %1 (excl)").arg(menu2->name));
216  }
217 
218  delete menu2;
219 }

222 VFolderMenu::insertSubMenu(SubMenu *parentMenu, const QString &menuName, SubMenu *newMenu, bool reversePriority)
223 {
224  const int i = menuName.indexOf('/');
225  const QString s1 = menuName.left(i);
226  const QString s2 = menuName.mid(i + 1);
227 
228  // Look up menu
229  foreach (SubMenu *menu, parentMenu->subMenus) {
230  if (menu->name == s1) {
231  if (i == -1) {
232  mergeMenu(menu, newMenu, reversePriority);
233  return;
234  } else {
235  insertSubMenu(menu, s2, newMenu, reversePriority);
236  return;
237  }
238  }
239  }
240  if (i == -1) {
241  // Add it here
242  newMenu->name = menuName;
243  parentMenu->subMenus.append(newMenu);
244  } else {
245  SubMenu *menu = new SubMenu;
246  menu->name = s1;
247  parentMenu->subMenus.append(menu);
248  insertSubMenu(menu, s2, newMenu);
249  }

510 void
511 VFolderMenu::mergeMenus(QDomElement &docElem, QString &name)
512 {
513  QMap<QString, QDomElement> menuNodes;
514  QMap<QString, QDomElement> directoryNodes;
515  QMap<QString, QDomElement> appDirNodes;
516  QMap<QString, QDomElement> directoryDirNodes;
517  QMap<QString, QDomElement> legacyDirNodes;
518  QDomElement defaultLayoutNode;
519  QDomElement layoutNode;
520 
521  QDomNode n = docElem.firstChild();
522  while (!n.isNull()) {
523  QDomElement e = n.toElement(); // try to convert the node to an element.
524  if (e.isNull()) {
525 // qDebug() << "Empty node";
526  } else if (e.tagName() == QLatin1String("DefaultAppDirs")) {
527  // Replace with m_defaultAppDirs
528  replaceNode(docElem, n, m_defaultAppDirs, QStringLiteral("AppDir"));
529  continue;
530  } else if (e.tagName() == QLatin1String("DefaultDirectoryDirs")) {
531  // Replace with m_defaultDirectoryDirs
532  replaceNode(docElem, n, m_defaultDirectoryDirs, QStringLiteral("DirectoryDir"));
533  continue;
534  } else if (e.tagName() == QLatin1String("DefaultMergeDirs")) {
535  // Replace with m_defaultMergeDirs
536  replaceNode(docElem, n, m_defaultMergeDirs, QStringLiteral("MergeDir"));
537  continue;
538  } else if (e.tagName() == QLatin1String("AppDir")) {
539  // Filter out dupes
540  foldNode(docElem, e, appDirNodes);
541  } else if (e.tagName() == QLatin1String("DirectoryDir")) {
542  // Filter out dupes
543  foldNode(docElem, e, directoryDirNodes);
544  } else if (e.tagName() == QLatin1String("LegacyDir")) {
545  // Filter out dupes
546  foldNode(docElem, e, legacyDirNodes);
547  } else if (e.tagName() == QLatin1String("Directory")) {
548  // Filter out dupes
549  foldNode(docElem, e, directoryNodes);
550  } else if (e.tagName() == QLatin1String("Move")) {
551  // Filter out dupes
552  QString orig;
553  QDomNode n2 = e.firstChild();
554  while (!n2.isNull()) {
555  QDomElement e2 = n2.toElement(); // try to convert the node to an element.
556  if (e2.tagName() == QLatin1String("Old")) {
557  orig = e2.text();
558  break;
559  }
560  n2 = n2.nextSibling();
561  }
562  foldNode(docElem, e, appDirNodes, orig);
563  } else if (e.tagName() == QLatin1String("Menu")) {
564  QString name;
565  mergeMenus(e, name);
566  QMap<QString, QDomElement>::iterator it = menuNodes.find(name);
567  if (it != menuNodes.end()) {
568  QDomElement docElem2 = *it;
569  QDomNode n2 = docElem2.firstChild();
570  QDomNode first = e.firstChild();
571  while (!n2.isNull()) {
572  QDomElement e2 = n2.toElement(); // try to convert the node to an element.
573  QDomNode n3 = n2.nextSibling();
574  e.insertBefore(n2, first);
575  docElem2.removeChild(n2);
576  n2 = n3;
577  }
578  // We still have duplicated Name entries
579  // but we don't care about that
580 
581  docElem.removeChild(docElem2);
582  menuNodes.erase(it);
583  }
584  menuNodes.insert(name, e);
585  } else if (e.tagName() == QLatin1String("MergeFile")) {
586  if ((e.attribute(QStringLiteral("type")) == QLatin1String("parent"))) {
587  // Ignore e.text(), as per the standard. We'll just look up the parent (more global) xml file.
588  pushDocInfoParent(e.attribute(QStringLiteral("__BasePath")), e.attribute(QStringLiteral("__BaseDir")));
589  } else {
590  pushDocInfo(e.text(), e.attribute(QStringLiteral("__BaseDir")));
591  }
592 
593  if (!m_docInfo.path.isEmpty()) {
594  mergeFile(docElem, n);
595  }
596  popDocInfo();
597 
598  QDomNode last = n;
599  n = n.nextSibling();
600  docElem.removeChild(last); // Remove the MergeFile node
601  continue;
602  } else if (e.tagName() == QLatin1String("MergeDir")) {
603  QString dir = absoluteDir(e.text(), e.attribute(QStringLiteral("__BaseDir")), true);
604  Q_ASSERT(dir.endsWith('/'));
605 
606  const bool relative = QDir::isRelativePath(dir);
607  const QStringList dirs = QStandardPaths::locateAll(QStandardPaths::GenericConfigLocation, QStringLiteral("menus/") + dir, QStandardPaths::LocateDirectory);
608  Q_FOREACH (const QString &menuDir, dirs) {
609  registerDirectory(menuDir);
610  }
611 
612  QStringList fileList;
613  Q_FOREACH (const QString &menuDir, dirs) {
614  const QStringList fileNames = QDir(menuDir).entryList(QStringList() << QStringLiteral("*.menu"));
615  Q_FOREACH (const QString &file, fileNames) {
616  const QString fileToAdd = relative ? dir + file : menuDir + file;
617  if (!fileList.contains(fileToAdd)) {
618  fileList.append(fileToAdd);
619  }
620  }
621  }
622 
623  Q_FOREACH (const QString &file, fileList) {
624  pushDocInfo(file);
625  mergeFile(docElem, n);
626  popDocInfo();
627  }
628 
629  QDomNode last = n;
630  n = n.nextSibling();
631  docElem.removeChild(last); // Remove the MergeDir node
632 
633  continue;
634  } else if (e.tagName() == QLatin1String("Name")) {
635  name = e.text();

827 void
828 VFolderMenu::processCondition(QDomElement &domElem, QHash<QString, KService::Ptr> &items)
829 {
830  if (domElem.tagName() == QLatin1String("And")) {
831  QDomNode n = domElem.firstChild();
832  // Look for the first child element
833  while (!n.isNull()) { // loop in case of comments
834  QDomElement e = n.toElement();
835  n = n.nextSibling();
836  if (!e.isNull()) {
837  processCondition(e, items);
838  break; // we only want the first one
839  }
840  }
841 
842  QHash<QString, KService::Ptr> andItems;
843  while (!n.isNull()) {
844  QDomElement e = n.toElement();
845  if (e.tagName() == QLatin1String("Not")) {
846  // Special handling for "and not"
847  QDomNode n2 = e.firstChild();
848  while (!n2.isNull()) {
849  QDomElement e2 = n2.toElement();
850  andItems.clear();
851  processCondition(e2, andItems);
852  excludeItems(items, andItems);
853  n2 = n2.nextSibling();
854  }
855  } else {
856  andItems.clear();
857  processCondition(e, andItems);
858  matchItems(items, andItems);
859  }
860  n = n.nextSibling();
861  }
862  } else if (domElem.tagName() == QLatin1String("Or")) {
863  QDomNode n = domElem.firstChild();
864  // Look for the first child element
865  while (!n.isNull()) { // loop in case of comments
866  QDomElement e = n.toElement();
867  n = n.nextSibling();
868  if (!e.isNull()) {
869  processCondition(e, items);
870  break; // we only want the first one
871  }
872  }
873 
874  QHash<QString, KService::Ptr> orItems;
875  while (!n.isNull()) {
876  QDomElement e = n.toElement();
877  if (!e.isNull()) {
878  orItems.clear();
879  processCondition(e, orItems);
880  includeItems(items, orItems);
881  }
882  n = n.nextSibling();
883  }
884  } else if (domElem.tagName() == QLatin1String("Not")) {
885  FOR_ALL_APPLICATIONS(it) {
886  KService::Ptr s = it.value();
887  items.insert(s->menuId(), s);
888  }
889  FOR_ALL_APPLICATIONS_END
890

892  QDomNode n = domElem.firstChild();
893  while (!n.isNull()) {
894  QDomElement e = n.toElement();
895  if (!e.isNull()) {
896  notItems.clear();
897  processCondition(e, notItems);
898  excludeItems(items, notItems);
899  }
900  n = n.nextSibling();
901  }
902  } else if (domElem.tagName() == QLatin1String("Category")) {
903  FOR_CATEGORY(domElem.text(), it) {
904  KService::Ptr s = *it;
905  items.insert(s->menuId(), s);
906  }
907  FOR_CATEGORY_END
908  } else if (domElem.tagName() == QLatin1String("All")) {
909  FOR_ALL_APPLICATIONS(it) {
910  KService::Ptr s = it.value();
911  items.insert(s->menuId(), s);
912  }
913  FOR_ALL_APPLICATIONS_END
914  } else if (domElem.tagName() == QLatin1String("Filename")) {
915  const QString filename = domElem.text();
916  //qDebug() << "Adding file" << filename;
917  KService::Ptr s = findApplication(filename);
918  if (s) {
919  items.insert(filename, s);
920  }
921  }

1002 VFolderMenu::processMenu(QDomElement &docElem, int pass)
1003 {

1006

1008  QString directoryFile;
1009  bool onlyUnallocated = false;
1010  bool isDeleted = false;
1011  QDomElement defaultLayoutNode;
1012  QDomElement layoutNode;
1013 
1014  QDomElement query;
1015  QDomNode n = docElem.firstChild();
1016  while (!n.isNull()) {
1017  QDomElement e = n.toElement(); // try to convert the node to an element.
1018  if (e.tagName() == QLatin1String("Name")) {
1019  name = e.text();
1020  } else if (e.tagName() == QLatin1String("Directory")) {
1021  directoryFile = e.text();
1022  } else if (e.tagName() == QLatin1String("DirectoryDir")) {
1023  QString dir = absoluteDir(e.text(), e.attribute(QStringLiteral("__BaseDir")));
1024 
1025  m_directoryDirs.prepend(dir);
1026  } else if (e.tagName() == QLatin1String("OnlyUnallocated")) {
1027  onlyUnallocated = true;
1028  } else if (e.tagName() == QLatin1String("NotOnlyUnallocated")) {
1029  onlyUnallocated = false;
1030  } else if (e.tagName() == QLatin1String("Deleted")) {
1031  isDeleted = true;
1032  } else if (e.tagName() == QLatin1String("NotDeleted")) {
1033  isDeleted = false;
1034  } else if (e.tagName() == QLatin1String("DefaultLayout")) {
1035  defaultLayoutNode = e;
1036  } else if (e.tagName() == QLatin1String("Layout")) {
1037  layoutNode = e;
1038  }
1039  n = n.nextSibling();
1040  }
1041 
1042  // Setup current menu entry
1043  if (pass == 0) {
1044  m_currentMenu = 0;
1045  // Look up menu
1046  if (parentMenu) {
1047  foreach (SubMenu *menu, parentMenu->subMenus) {
1048  if (menu->name == name) {
1049  m_currentMenu = menu;
1050  break;
1051  }
1052  }
1053  }
1054 
1055  if (!m_currentMenu) { // Not found?
1056  // Create menu
1057  m_currentMenu = new SubMenu;
1058  m_currentMenu->name = name;
1059 
1060  if (parentMenu) {
1061  parentMenu->subMenus.append(m_currentMenu);
1062  } else {
1063  m_rootMenu = m_currentMenu;
1064  }
1065  }
1066  if (directoryFile.isEmpty()) {
1067  //qDebug() << "Menu" << name << "does not specify a directory file.";
1068  }
1069 
1070  // Override previous directoryFile iff available
1071  QString tmp = locateDirectoryFile(directoryFile);
1072  if (! tmp.isEmpty()) {
1073  m_currentMenu->directoryFile = tmp;
1074  }
1075  m_currentMenu->isDeleted = isDeleted;
1076 
1077  m_currentMenu->defaultLayoutNode = defaultLayoutNode;
1078  m_currentMenu->layoutNode = layoutNode;
1079  } else {
1080  // Look up menu
1081  if (parentMenu) {
1082  foreach (SubMenu *menu, parentMenu->subMenus) {
1083  if (menu->name == name) {
1084  m_currentMenu = menu;
1085  break;
1086  }
1087  }
1088  } else {
1089  m_currentMenu = m_rootMenu;
1090  }
1091  }
1092 
1093  // Process AppDir and LegacyDir
1094  if (pass == 0) {
1095  QDomElement query;
1096  QDomNode n = docElem.firstChild();
1097  while (!n.isNull()) {
1098  QDomElement e = n.toElement(); // try to convert the node to an element.
1099  if (e.tagName() == QLatin1String("AppDir")) {
1100  createAppsInfo();

1102 
1103  registerDirectory(dir);
1104 
1105  loadApplications(dir, QString());

1116 
1117  processLegacyDir(dir, QString(), prefix);
1118 
1119  m_legacyNodes.insert(dir, m_currentMenu);
1120  m_currentMenu = oldMenu;
1121  }
1122  n = n.nextSibling();
1123  }
1124  }
1125 
1126  loadAppsInfo(); // Update the scope wrt the list of applications
1127 
1128  if (((pass == 1) && !onlyUnallocated) || ((pass == 2) && onlyUnallocated)) {
1129  n = docElem.firstChild();

1230  }
1231 
1232  }
1233

1243 static QString parseAttribute(const QDomElement &e)
1244 {
1245  QString option;
1246 
1247  const QString SHOW_EMPTY=QStringLiteral("show_empty");
1248  if (e.hasAttribute(SHOW_EMPTY)) {
1249  QString str = e.attribute(SHOW_EMPTY);
1250  if (str == QLatin1String("true")) {
1251  option = "ME ";
1252  } else if (str == QLatin1String("false")) {
1253  option = "NME ";
1254  } else {
1255  //qDebug()<<" Error in parsing show_empty attribute :"<<str;
1256  }
1257  }
1258  const QString INLINE=QStringLiteral("inline");
1259  if (e.hasAttribute(INLINE)) {
1260  QString str = e.attribute(INLINE);
1261  if (str == QLatin1String("true")) {
1262  option += "I ";
1263  } else if (str == QLatin1String("false")) {
1264  option += "NI ";
1265  } else {
1266  qDebug() << " Error in parsing inline attribute :" << str;
1267  }
1268  }
1269  if (e.hasAttribute(QStringLiteral("inline_limit"))) {
1270  bool ok;
1271  int value = e.attribute(QStringLiteral("inline_limit")).toInt(&ok);
1272  if (ok) {
1273  option += QStringLiteral("IL[%1] ").arg(value);
1274  }
1275  }
1276  if (e.hasAttribute(QStringLiteral("inline_header"))) {
1277  QString str = e.attribute(QStringLiteral("inline_header"));
1278  if (str == QLatin1String("true")) {
1279  option += QStringLiteral("IH ");
1280  } else if (str == QLatin1String("false")) {
1281  option += QStringLiteral("NIH ");
1282  } else {
1283  qDebug() << " Error in parsing of inline_header attribute :" << str;
1284  }
1285 
1286  }
1287  if (e.hasAttribute(QStringLiteral("inline_alias")) && e.attribute(QStringLiteral("inline_alias")) == QLatin1String("true")) {
1288  QString str = e.attribute(QStringLiteral("inline_alias"));
1289  if (str == QLatin1String("true")) {
1290  option += "IA";
1291  } else if (str == QLatin1String("false")) {

1304 static QStringList parseLayoutNode(const QDomElement &docElem)
1305 {
1306  QStringList layout;
1307 
1308  QString optionDefaultLayout;
1309  if (docElem.tagName() == QLatin1String("DefaultLayout")) {
1310  optionDefaultLayout = parseAttribute(docElem);
1311  }
1312  if (!optionDefaultLayout.isEmpty()) {
1313  layout.append(optionDefaultLayout);
1314  }
1315 
1316  bool mergeTagExists = false;
1317  QDomNode n = docElem.firstChild();
1318  while (!n.isNull()) {
1319  QDomElement e = n.toElement(); // try to convert the node to an element.
1320  if (e.tagName() == QLatin1String("Separator")) {
1321  layout.append(QStringLiteral(":S"));
1322  } else if (e.tagName() == QLatin1String("Filename")) {
1323  layout.append(e.text());
1324  } else if (e.tagName() == QLatin1String("Menuname")) {
1325  layout.append('/' + e.text());
1326  QString option = parseAttribute(e);
1327  if (!option.isEmpty()) {
1328  layout.append(option);
1329  }
1330  } else if (e.tagName() == QLatin1String("Merge")) {
1331  QString type = e.attributeNode(QStringLiteral("type")).value();
1332  if (type == QLatin1String("files")) {
1333  layout.append(QStringLiteral(":F"));
1334  } else if (type == QLatin1String("menus")) {
1335  layout.append(QStringLiteral(":M"));
1336  } else if (type == QLatin1String("all")) {
1337  layout.append(QStringLiteral(":A"));
1338  }
1339  mergeTagExists = true;
1340  }
1341 
1342  n = n.nextSibling();
1343  }
1344 
1345  if (!mergeTagExists) {
1346  layout.append(QStringLiteral(":M"));
1347  layout.append(QStringLiteral(":F"));
1348  qCWarning(SYCOCA) << "The menu spec file contains a Layout or DefaultLayout tag without the mandatory Merge tag inside. Please fix your file.";
1349  }
1350  return layout;
1351 }
