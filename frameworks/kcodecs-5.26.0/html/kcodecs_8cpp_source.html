167 // parse the encoded-word (scursor points to after the initial '=')
168 bool parseEncodedWord(const char *&scursor, const char *const send,
169  QString *result, QByteArray *language,
170  QByteArray *usedCS, const QByteArray &defaultCS,
171  CharsetOption charsetOption)
172 {
173  assert(result);
174  assert(language);
175 
176  // make sure the caller already did a bit of the work.
177  assert(*(scursor - 1) == '=');
178 
179  //
180  // STEP 1:
181  // scan for the charset/language portion of the encoded-word
182  //
183 
184  char ch = *scursor++;
185 
186  if (ch != '?') {
187  // qDebug() << "first";
188  // qDebug() << "Premature end of encoded word";
189  return false;
190  }
191 
192  // remember start of charset (ie. just after the initial "=?") and
193  // language (just after the first '*') fields:
194  const char *charsetStart = scursor;
195  const char *languageStart = 0;
196 
197  // find delimiting '?' (and the '*' separating charset and language
198  // tags, if any):
199  for (; scursor != send ; scursor++) {
200  if (*scursor == '?') {
201  break;
202  } else if (*scursor == '*' && languageStart == 0) {
203  languageStart = scursor + 1;
204  }
205  }
206 
207  // not found? can't be an encoded-word!
208  if (scursor == send || *scursor != '?') {
209  // qDebug() << "second";
210  // qDebug() << "Premature end of encoded word";
211  return false;
212  }
213 
214  // extract the language information, if any (if languageStart is 0,
215  // language will be null, too):
216  QByteArray maybeLanguage(languageStart, scursor - languageStart);
217  // extract charset information (keep in mind: the size given to the
218  // ctor is one off due to the \0 terminator):
219  QByteArray maybeCharset(charsetStart,
220  (languageStart ? languageStart - 1 : scursor) - charsetStart);
221 
222  //
223  // STEP 2:
224  // scan for the encoding portion of the encoded-word
225  //
226 
227  // remember start of encoding (just _after_ the second '?'):
228  scursor++;
229  const char *encodingStart = scursor;
230 
231  // find next '?' (ending the encoding tag):
232  for (; scursor != send ; scursor++) {
233  if (*scursor == '?') {
234  break;
235  }
236  }
237 
238  // not found? Can't be an encoded-word!
239  if (scursor == send || *scursor != '?') {
240  // qDebug() << "third";
241  // qDebug() << "Premature end of encoded word";
242  return false;
243  }
244 
245  // extract the encoding information:
246  QByteArray maybeEncoding(encodingStart, scursor - encodingStart);
247 
248  // qDebug() << "parseEncodedWord: found charset == \"" << maybeCharset
249  // << "\"; language == \"" << maybeLanguage
250  // << "\"; encoding == \"" << maybeEncoding << "\"";
251 
252  //
253  // STEP 3:
254  // scan for encoded-text portion of encoded-word
255  //
256 
257  // remember start of encoded-text (just after the third '?'):
258  scursor++;
259  const char *encodedTextStart = scursor;
260 
261  // find the '?=' sequence (ending the encoded-text):
262  for (; scursor != send ; scursor++) {
263  if (*scursor == '?') {
264  if (scursor + 1 != send) {
265  if (*(scursor + 1) != '=') { // We expect a '=' after the '?', but we got something else; ignore
266  // qDebug() << "Stray '?' in q-encoded word, ignoring this.";
267  continue;
268  } else { // yep, found a '?=' sequence
269  scursor += 2;
270  break;
271  }
272  } else { // The '?' is the last char, but we need a '=' after it!
273  // qDebug() << "Premature end of encoded word";
274  return false;
275  }
276  }
277  }
278 
279  if (*(scursor - 2) != '?' || *(scursor - 1) != '=' || scursor < encodedTextStart + 2) {
280  // qDebug() << "Premature end of encoded word";
281  return false;
282  }
283 
284  // set end sentinel for encoded-text:
285  const char *const encodedTextEnd = scursor - 2;
286 
287  //
288  // STEP 4:
289  // setup decoders for the transfer encoding and the charset
290  //
291 
292  // try if there's a codec for the encoding found:
293  Codec *codec = Codec::codecForName(maybeEncoding);
294  if (!codec) {
295  // qDebug() << "Unknown encoding" << maybeEncoding;
296  return false;
297  }
298 
299  // get an instance of a corresponding decoder:
300  Decoder *dec = codec->makeDecoder();
301  assert(dec);
302 
303  // try if there's a (text)codec for the charset found:
304  bool matchOK = false;
305  QByteArray cs;
306  QTextCodec *textCodec = 0;
307  if (charsetOption == KCodecs::ForceDefaultCharset || maybeCharset.isEmpty()) {
308  textCodec = KCharsets::charsets()->codecForName(QLatin1String(defaultCS), matchOK);
309  cs = cachedCharset(defaultCS);
310  } else {
311  textCodec = KCharsets::charsets()->codecForName(QLatin1String(maybeCharset), matchOK);
312  if (!matchOK) { //no suitable codec found => use default charset
313  textCodec = KCharsets::charsets()->codecForName(QLatin1String(defaultCS), matchOK);
314  cs = cachedCharset(defaultCS);
315  } else {
316  cs = cachedCharset(maybeCharset);
317  }
318  }
319  if (usedCS) {
320  *usedCS = cs;
321  }
322 
323  if (!matchOK || !textCodec) {
324  // qDebug() << "Unknown charset" << maybeCharset;
325  delete dec;
326  return false;
327  };
328 
329  // qDebug() << "mimeName(): \"" << textCodec->name() << "\"";
330 
331  // allocate a temporary buffer to store the 8bit text:
332  int encodedTextLength = encodedTextEnd - encodedTextStart;
333  QByteArray buffer;
334  buffer.resize(codec->maxDecodedSizeFor(encodedTextLength));
335  char *bbegin = buffer.data();
336  char *bend = bbegin + buffer.length();
337 
338  //
339  // STEP 5:
340  // do the actual decoding
341  //
342 
343  if (!dec->decode(encodedTextStart, encodedTextEnd, bbegin, bend)) {
344  qWarning() << codec->name() << "codec lies about its maxDecodedSizeFor("
345  << encodedTextLength << ")\nresult may be truncated";
346  }
347 
348  *result = textCodec->toUnicode(buffer.data(), bbegin - buffer.data());
349 
350  // qDebug() << "result now: \"" << result << "\"";
351  // cleanup:
352  delete dec;
353  *language = maybeLanguage;
354 
355  return true;
356 }

368 QString KCodecs::decodeRFC2047String(const QByteArray &src, QByteArray *usedCS,
369  const QByteArray &defaultCS,
370  CharsetOption charsetOption)
371 {
372  QByteArray result;
373  QByteArray spaceBuffer;
374  const char *scursor = src.constData();
375  const char *send = scursor + src.length();
376  bool onlySpacesSinceLastWord = false;
377 
378  while (scursor != send) {
379  // space
380  if (isspace(*scursor) && onlySpacesSinceLastWord) {
381  spaceBuffer += *scursor++;
382  continue;
383  }
384 
385  // possible start of an encoded word
386  if (*scursor == '=') {
387  QByteArray language;
388  QString decoded;
389  ++scursor;
390  const char *start = scursor;
391  if (parseEncodedWord(scursor, send, &decoded, &language, usedCS, defaultCS, charsetOption)) {
392  result += decoded.toUtf8();
393  onlySpacesSinceLastWord = true;
394  spaceBuffer.clear();
395  } else {
396  if (onlySpacesSinceLastWord) {
397  result += spaceBuffer;
398  onlySpacesSinceLastWord = false;
399  }
400  result += '=';
401  scursor = start; // reset cursor after parsing failure
402  }
403  continue;
404  } else {
405  // unencoded data
406  if (onlySpacesSinceLastWord) {
407  result += spaceBuffer;
408  onlySpacesSinceLastWord = false;
409  }
410  result += *scursor;
411  ++scursor;
412  }
413  }
414  // If there are any chars that couldn't be decoded in UTF-8,
415  // fallback to local codec
416  const QString tryUtf8 = QString::fromUtf8(result);
417  if (tryUtf8.contains(0xFFFD)) {
418  QTextCodec *codec = QTextCodec::codecForLocale();
419  return codec->toUnicode(result);
420  } else {
421  return tryUtf8;
422  }
423 }

426 QByteArray KCodecs::encodeRFC2047String(const QString &src, const QByteArray &charset)
427 {
428  QByteArray result;
429  int start = 0, end = 0;
430  bool nonAscii = false, ok = true, useQEncoding = false;
431 
432  // fromLatin1() is safe here, codecForName() uses toLatin1() internally
433  const QTextCodec *codec = KCharsets::charsets()->codecForName(QString::fromLatin1(charset), ok);
434 
435  QByteArray usedCS;
436  if (!ok) {
437  //no codec available => try local8Bit and hope the best ;-)
438  codec = QTextCodec::codecForLocale();
439  usedCS = codec->name();
440  } else {
441  Q_ASSERT(codec);
442  if (charset.isEmpty()) {
443  usedCS = codec->name();
444  } else {
445  usedCS = charset;
446  }
447  }
448 
449  QTextCodec::ConverterState converterState(QTextCodec::IgnoreHeader);
450  QByteArray encoded8Bit = codec->fromUnicode(src.constData(), src.length(), &converterState);
451  if (converterState.invalidChars > 0) {
452  usedCS = "utf-8";
453  codec = QTextCodec::codecForName(usedCS);
454  encoded8Bit = codec->fromUnicode(src);
455  }
456 
457  if (usedCS.contains("8859-")) { // use "B"-Encoding for non iso-8859-x charsets
458  useQEncoding = true;
459  }
460 
461  uint encoded8BitLength = encoded8Bit.length();
462  for (unsigned int i = 0; i < encoded8BitLength; i++) {
463  if (encoded8Bit[i] == ' ') { // encoding starts at word boundaries
464  start = i + 1;
465  }
466 
467  // encode escape character, for japanese encodings...
468  if (((signed char)encoded8Bit[i] < 0) || (encoded8Bit[i] == '\033')) {
469  end = start; // non us-ascii char found, now we determine where to stop encoding
470  nonAscii = true;
471  break;
472  }
473  }
474 
475  if (nonAscii) {
476  while ((end < encoded8Bit.length()) && (encoded8Bit[end] != ' ')) {
477  // we encode complete words
478  end++;
479  }
480 
481  for (int x = end; x < encoded8Bit.length(); x++) {
482  if (((signed char)encoded8Bit[x] < 0) || (encoded8Bit[x] == '\033')) {
483  end = x; // we found another non-ascii word
484 
485  while ((end < encoded8Bit.length()) && (encoded8Bit[end] != ' ')) {
486  // we encode complete words
487  end++;
488  }
489  }
490  }
491 
492  result = encoded8Bit.left(start) + "=?" + usedCS;
493 
494  if (useQEncoding) {
495  result += "?Q?";
496 
497  char c, hexcode;// "Q"-encoding implementation described in RFC 2047
498  for (int i = start; i < end; i++) {
499  c = encoded8Bit[i];
500  if (c == ' ') { // make the result readable with not MIME-capable readers
501  result += '_';
502  } else {
503  if (((c >= 'a') && (c <= 'z')) || // paranoid mode, encode *all* special chars to avoid problems
504  ((c >= 'A') && (c <= 'Z')) || // with "From" & "To" headers
505  ((c >= '0') && (c <= '9'))) {
506  result += c;
507  } else {
508  result += '='; // "stolen" from KMail ;-)
509  hexcode = ((c & 0xF0) >> 4) + 48;
510  if (hexcode >= 58) {
511  hexcode += 7;
512  }
513  result += hexcode;
514  hexcode = (c & 0x0F) + 48;
515  if (hexcode >= 58) {
516  hexcode += 7;
517  }
518  result += hexcode;
519  }
520  }
521  }
522  } else {
523  result += "?B?" + encoded8Bit.mid(start, end - start).toBase64();
524  }
525 
526  result += "?=";
527  result += encoded8Bit.right(encoded8Bit.length() - end);
528  } else {
529  result = encoded8Bit;
530  }
531 
532  return result;
533 }

594 bool KCodecs::Codec::encode(const char *&scursor, const char *const send,
595  char *&dcursor, const char *const dend,
596  NewlineType newline) const
597 {
598  // get an encoder:
599  QScopedPointer<Encoder> enc(makeEncoder(newline));
600  assert(!enc.isNull());
601 
602  // encode and check for output buffer overflow:
603  while (!enc->encode(scursor, send, dcursor, dend)) {
604  if (dcursor == dend) {
605  return false; // not enough space in output buffer
606  }
607  }
608 
609  // finish and check for output buffer overflow:
610  while (!enc->finish(dcursor, dend)) {
611  if (dcursor == dend) {
612  return false; // not enough space in output buffer
613  }
614  }
615 
616  return true; // successfully encoded.
617 }

619 QByteArray KCodecs::Codec::encode(const QByteArray &src, NewlineType newline) const
620 {
621  // allocate buffer for the worst case:
622  QByteArray result;
623  result.resize(maxEncodedSizeFor(src.size(), newline));
624 
625  // set up iterators:
626  QByteArray::ConstIterator iit = src.begin();
627  QByteArray::ConstIterator iend = src.end();
628  QByteArray::Iterator oit = result.begin();
629  QByteArray::ConstIterator oend = result.end();
630 
631  // encode
632  if (!encode(iit, iend, oit, oend, newline)) {
633  qCritical() << name() << "codec lies about it's mEncodedSizeFor()";
634  }
635 
636  // shrink result to actual size:
637  result.truncate(oit - result.begin());
638 
639  return result;

665 bool KCodecs::Codec::decode(const char *&scursor, const char *const send,
666  char *&dcursor, const char *const dend,
667  NewlineType newline) const
668 {
669  // get a decoder:
670  QScopedPointer<Decoder> dec(makeDecoder(newline));
671  assert(!dec.isNull());
672 
673  // decode and check for output buffer overflow:
674  while (!dec->decode(scursor, send, dcursor, dend)) {
675  if (dcursor == dend) {
676  return false; // not enough space in output buffer
677  }
678  }

710 bool KCodecs::Encoder::write(char ch, char*& dcursor, const char*const dend)
711 {
712  if (dcursor != dend) {
713  // if there's space in the output stream, write there:
714  *dcursor++ = ch;
715  return true;
716  } else {
717  // else buffer the output:
718  if (d->outputBufferCursor >= maxBufferedChars) {
719  qCritical()
720  << "KCodecs::Encoder: internal buffer overflow!";
721  } else {
722  d->outputBuffer[ d->outputBufferCursor++ ] = ch;
723  }
724  return false;
725  }
726 }

728 // write as much as possible off the output buffer. Return true if
729 // flushing was complete, false if some chars could not be flushed.
730 bool KCodecs::Encoder::flushOutputBuffer(char *&dcursor, const char *const dend)
731 {
732  int i;
733  // copy output buffer to output stream:
734  for (i = 0 ; dcursor != dend && i < d->outputBufferCursor ; ++i) {
735  *dcursor++ = d->outputBuffer[i];
736  }
737 
738  // calculate the number of missing chars:
739  int numCharsLeft = d->outputBufferCursor - i;
740  // push the remaining chars to the begin of the buffer:
741  if (numCharsLeft) {
742  ::memmove(d->outputBuffer, d->outputBuffer + i, numCharsLeft);
743  }
744  // adjust cursor:
745  d->outputBufferCursor = numCharsLeft;
746 
747  return !numCharsLeft;
748 }
