503 void KPropertiesDialog::accept()
504 {
505  QList<KPropertiesDialogPlugin *>::const_iterator pageListIt;
506  d->m_aborted = false;
507 
508  KFilePropsPlugin *filePropsPlugin = qobject_cast<KFilePropsPlugin *>(d->m_pageList.first());
509 
510  // If any page is dirty, then set the main one (KFilePropsPlugin) as
511  // dirty too. This is what makes it possible to save changes to a global
512  // desktop file into a local one. In other cases, it doesn't hurt.
513  for (pageListIt = d->m_pageList.constBegin(); pageListIt != d->m_pageList.constEnd(); ++pageListIt) {
514  if ((*pageListIt)->isDirty() && filePropsPlugin) {
515  filePropsPlugin->setDirty();
516  break;
517  }
518  }
519 
520  // Apply the changes in the _normal_ order of the tabs now
521  // This is because in case of renaming a file, KFilePropsPlugin will call
522  // KPropertiesDialog::rename, so other tab will be ok with whatever order
523  // BUT for file copied from templates, we need to do the renaming first !
524  for (pageListIt = d->m_pageList.constBegin(); pageListIt != d->m_pageList.constEnd() && !d->m_aborted; ++pageListIt) {
525  if ((*pageListIt)->isDirty()) {
526  // qDebug() << "applying changes for " << (*pageListIt)->metaObject()->className();
527  (*pageListIt)->applyChanges();
528  // applyChanges may change d->m_aborted.
529  } else {
530  // qDebug() << "skipping page " << (*pageListIt)->metaObject()->className();
531  }
532  }
533 
534  if (!d->m_aborted && filePropsPlugin) {
535  filePropsPlugin->postApplyChanges();
536  }
537 
538  if (!d->m_aborted) {
539  emit applied();
540  emit propertiesClosed();
541  deleteLater(); // somewhat like Qt::WA_DeleteOnClose would do.
542  KPageDialog::accept();
543  } // else, keep dialog open for user to fix the problem.
544 }
