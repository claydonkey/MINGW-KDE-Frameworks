97 QStringList FakeManager::devicesFromQuery(const QString &parentUdi, Solid::DeviceInterface::Type type)
98 {
99  if (!parentUdi.isEmpty()) {
100  QStringList found = findDeviceStringMatch(QLatin1String("parent"), parentUdi);
101 
102  if (type == Solid::DeviceInterface::Unknown) {
103  return found;
104  }
105 
106  QStringList result;
107 
108  QStringList::ConstIterator it = found.constBegin();
109  QStringList::ConstIterator end = found.constEnd();
110 
111  for (; it != end; ++it) {
112  FakeDevice *device = d->loadedDevices[*it];
113 
114  if (device->queryDeviceInterface(type)) {
115  result << *it;
116  }
117  }
118 
119  return result;
120  } else if (type != Solid::DeviceInterface::Unknown) {
121  return findDeviceByDeviceInterface(type);
122  } else {
123  return allDevices();
124  }
125 }

190 void FakeManager::parseMachineFile()
191 {
192  QFile machineFile(d->xmlFile);
193  if (!machineFile.open(QIODevice::ReadOnly)) {
194  qWarning() << Q_FUNC_INFO << "Error while opening " << d->xmlFile << endl;
195  return;
196  }
197 
198  QDomDocument fakeDocument;
199  if (!fakeDocument.setContent(&machineFile)) {
200  qWarning() << Q_FUNC_INFO << "Error while creating the QDomDocument." << endl;
201  machineFile.close();
202  return;
203  }
204  machineFile.close();
205 
206  qDebug() << Q_FUNC_INFO << "Parsing fake computer XML: " << d->xmlFile << endl;
207  QDomElement mainElement = fakeDocument.documentElement();
208  QDomNode node = mainElement.firstChild();
209  while (!node.isNull()) {
210  QDomElement tempElement = node.toElement();
211  if (!tempElement.isNull() && tempElement.tagName() == QLatin1String("device")) {
212  FakeDevice *tempDevice = parseDeviceElement(tempElement);
213  if (tempDevice) {
214  Q_ASSERT(!d->loadedDevices.contains(tempDevice->udi()));
215  d->loadedDevices.insert(tempDevice->udi(), tempDevice);
216  emit deviceAdded(tempDevice->udi());
217  }
218  }
219 
220  node = node.nextSibling();
221  }
222 }

224 FakeDevice *FakeManager::parseDeviceElement(const QDomElement &deviceElement)
225 {
226  FakeDevice *device = 0;
227  QMap<QString, QVariant> propertyMap;
228  QString udi = deviceElement.attribute("udi");
229 
230  QDomNode propertyNode = deviceElement.firstChild();
231  while (!propertyNode.isNull()) {
232  QDomElement propertyElement = propertyNode.toElement();
233  if (!propertyElement.isNull() && propertyElement.tagName() == QLatin1String("property")) {
234  QString propertyKey;
235  QVariant propertyValue;
236 
237  propertyKey = propertyElement.attribute("key");
238  propertyValue = QVariant(propertyElement.text());
239 
240  propertyMap.insert(propertyKey, propertyValue);
241  }
242 
243  propertyNode = propertyNode.nextSibling();
244  }
245 
246  if (!propertyMap.isEmpty()) {
247  device = new FakeDevice(udi, propertyMap);
248  }
249 
250  return device;
251 }
