58 bool OpticalDrive::eject()
59 {
60  if (m_ejectInProgress) {
61  return false;
62  }
63  m_ejectInProgress = true;
64  m_device->broadcastActionRequested("eject");
65 
66  const QString path = m_device->udi();
67  QDBusConnection c = QDBusConnection::connectToBus(QDBusConnection::SystemBus, "Solid::Udisks2::OpticalDrive::" + path);
68 
69  // if the device is mounted, unmount first
70  QString blockPath;
71  org::freedesktop::DBus::ObjectManager manager(UD2_DBUS_SERVICE, UD2_DBUS_PATH, c);
72  QDBusPendingReply<DBUSManagerStruct> reply = manager.GetManagedObjects();
73  reply.waitForFinished();
74  if (!reply.isError()) { // enum devices
75  Q_FOREACH (const QDBusObjectPath &objPath, reply.value().keys()) {
76  const QString udi = objPath.path();
77 
78  //qDebug() << "Inspecting" << udi;
79 
80  if (udi == UD2_DBUS_PATH_MANAGER || udi == UD2_UDI_DISKS_PREFIX || udi.startsWith(UD2_DBUS_PATH_JOBS)) {
81  continue;
82  }
83 
84  Device device(udi);
85  if (device.drivePath() == path && device.isMounted()) {
86  //qDebug() << "Got mounted block device:" << udi;
87  blockPath = udi;
88  break;
89  }
90  }
91  } else { // show error
92  qWarning() << "Failed enumerating UDisks2 objects:" << reply.error().name() << "\n" << reply.error().message();
93  }
94 
95  if (!blockPath.isEmpty()) {
96  //qDebug() << "Calling unmount on" << blockPath;
97  QDBusMessage msg = QDBusMessage::createMethodCall(UD2_DBUS_SERVICE, blockPath, UD2_DBUS_INTERFACE_FILESYSTEM, "Unmount");
98  msg << QVariantMap(); // options, unused now
99  c.call(msg, QDBus::BlockWithGui);
100  }
101 
102  QDBusMessage msg = QDBusMessage::createMethodCall(UD2_DBUS_SERVICE, path, UD2_DBUS_INTERFACE_DRIVE, "Eject");
103  msg << QVariantMap();
104  return c.callWithCallback(msg, this, SLOT(slotDBusReply(QDBusMessage)), SLOT(slotDBusError(QDBusError)));
105 }
