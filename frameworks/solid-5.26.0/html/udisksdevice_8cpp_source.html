48 static QString formatByteSize(double size)
49 {
50  // Per IEC 60027-2
51 
52  // Binary prefixes
53  //Tebi-byte TiB 2^40 1,099,511,627,776 bytes
54  //Gibi-byte GiB 2^30 1,073,741,824 bytes
55  //Mebi-byte MiB 2^20 1,048,576 bytes
56  //Kibi-byte KiB 2^10 1,024 bytes
57 
58  QString s;
59  // Gibi-byte
60  if (size >= 1073741824.0) {
61  size /= 1073741824.0;
62  if (size > 1024) { // Tebi-byte
63  s = QCoreApplication::translate("udisksdevice", "%1 TiB").arg(QLocale().toString(size / 1024.0, 'f', 1));
64  } else {
65  s = QCoreApplication::translate("udisksdevice", "%1 GiB").arg(QLocale().toString(size, 'f', 1));
66  }
67  }
68  // Mebi-byte
69  else if (size >= 1048576.0) {
70  size /= 1048576.0;
71  s = QCoreApplication::translate("udisksdevice", "%1 MiB").arg(QLocale().toString(size, 'f', 1));
72  }
73  // Kibi-byte
74  else if (size >= 1024.0) {
75  size /= 1024.0;
76  s = QCoreApplication::translate("udisksdevice", "%1 KiB").arg(QLocale().toString(size, 'f', 1));
77  }
78  // Just byte
79  else if (size > 0) {
80  s = QCoreApplication::translate("udisksdevice", "%1 B").arg(QLocale().toString(size, 'f', 1));
81  }
82  // Nothing
83  else {
84  s = QCoreApplication::translate("udisksdevice", "0 B");
85  }
86  return s;
87 }

159 QObject *Device::createDeviceInterface(const Solid::DeviceInterface::Type &type)
160 {
161  if (!queryDeviceInterface(type)) {
162  return 0;
163  }
164 
165  DeviceInterface *iface = 0;
166  switch (type) {
167  case Solid::DeviceInterface::GenericInterface:

173  case Solid::DeviceInterface::StorageAccess:

176  case Solid::DeviceInterface::StorageDrive:

179  case Solid::DeviceInterface::OpticalDrive:
180  iface = new OpticalDrive(this);
181  break;
182  case Solid::DeviceInterface::StorageVolume:
183  iface = new StorageVolume(this);
184  break;
185  case Solid::DeviceInterface::OpticalDisc:
186  iface = new OpticalDisc(this);
187  break;

587 QString Device::icon() const
588 {
589  QString iconName = property("HintIconName").toString(); // non-cached
590 
591  if (!iconName.isEmpty()) {
592  return iconName;
593  } else if (isLoop() || isSwap()) {
594  return "drive-harddisk";
595  } else if (isDrive()) {
596  const bool isRemovable = prop("Removable").toBool();
597  const QString conn = prop("ConnectionBus").toString();
598 
599  if (isOpticalDrive()) {
600  return "drive-optical";
601  } else if (isRemovable && !prop("Optical").toBool()) {
602  if (conn == "usb") {
603  return "drive-removable-media-usb";
604  } else {
605  return "drive-removable-media";
606  }
607  }
608  } else if (isBlock()) {
609  const QString drv = drivePath();
610  if (drv.isEmpty() || drv == "/") {
611  return "drive-harddisk"; // stuff like loop devices or swap which don't have the Drive prop set
612  }
613 
614  Device drive(drv);
615 
616  // handle media
617  const QString media = drive.prop("Media").toString();
618 
619  if (!media.isEmpty()) {
620  if (drive.prop("Optical").toBool()) { // optical stuff
621  bool isWritable = drive.prop("OpticalBlank").toBool();
622 
623  const UDisks2::OpticalDisc disc(const_cast<Device *>(this));
624  Solid::OpticalDisc::ContentTypes availContent = disc.availableContent();
625 
626  if (availContent & Solid::OpticalDisc::VideoDvd) { // Video DVD
627  return "media-optical-dvd-video";
628  } else if ((availContent & Solid::OpticalDisc::VideoCd) || (availContent & Solid::OpticalDisc::SuperVideoCd)) { // Video CD
629  return "media-optical-video";
630  } else if ((availContent & Solid::OpticalDisc::Data) && (availContent & Solid::OpticalDisc::Audio)) { // Mixed CD
631  return "media-optical-mixed-cd";
632  } else if (availContent & Solid::OpticalDisc::Audio) { // Audio CD
633  return "media-optical-audio";
634  } else if (availContent & Solid::OpticalDisc::Data) { // Data CD
635  return "media-optical-data";
636  } else if (isWritable) {
637  return "media-optical-recordable";
638  } else {
639  if (media.startsWith("optical_dvd") || media.startsWith("optical_hddvd")) { // DVD
640  return "media-optical-dvd";
641  } else if (media.startsWith("optical_bd")) { // BluRay
642  return "media-optical-blu-ray";
643  }
644  }
645 
646  // fallback for every other optical disc
647  return "media-optical";
648  }
649 
650  if (media == "flash_ms") { // Flash & Co.
651  return "media-flash-memory-stick";
652  } else if (media == "flash_sd" || media == "flash_sdhc" || media == "flash_sdxc" || media == "flash_mmc") {
653  return "media-flash-sd-mmc";
654  } else if (media == "flash_sm") {
655  return "media-flash-smart-media";
656  } else if (media == "thumb") {
657  return "drive-removable-media-usb-pendrive";
658  } else if (media.startsWith("flash")) {
659  return "media-flash";
660  } else if (media == "floppy") { // the good ol' floppy
661  return "media-floppy";
662  }
663  }
664 
665  if (drive.prop("ConnectionBus").toString() == "sdio") { // hack for SD cards connected thru sdio bus
666  return "media-flash-sd-mmc";
667  }
668 
669  return drive.icon();
670  }
671 
672  return "drive-harddisk"; // general fallback
673 }

676 {
