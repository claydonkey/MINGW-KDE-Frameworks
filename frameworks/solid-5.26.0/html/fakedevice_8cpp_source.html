45 FakeDevice::FakeDevice(const QString &udi, const QMap<QString, QVariant> &propertyMap)
46  : Solid::Ifaces::Device(), d(new Private)
47 {
48  d->udi = udi;
49  d->propertyMap = propertyMap;
50  d->interfaceList = d->propertyMap["interfaces"].toString().simplified().split(',');
51  d->interfaceList << "GenericInterface";
52  d->locked = false;
53  d->broken = false;
54 
55  QDBusConnection::sessionBus().registerObject(udi, this, QDBusConnection::ExportNonScriptableSlots);
56 
57  // Force instantiation of all the device interfaces
58  // this way they'll get exported on the bus
59  // that means they'll be created twice, but that won't be
60  // a problem for unit testing.
61  Q_FOREACH (const QString &interface, d->interfaceList) {
62  Solid::DeviceInterface::Type type = Solid::DeviceInterface::stringToType(interface);
63  createDeviceInterface(type);
64  }
65 
66  connect(d.data(), SIGNAL(propertyChanged(QMap<QString,int>)),
67  this, SIGNAL(propertyChanged(QMap<QString,int>)));
68  connect(d.data(), SIGNAL(conditionRaised(QString,QString)),
69  this, SIGNAL(conditionRaised(QString,QString)));
70 }

162 bool FakeDevice::setProperty(const QString &key, const QVariant &value)
163 {
164  if (d->broken) {
165  return false;
166  }
167 
168  Solid::GenericInterface::PropertyChange change_type = Solid::GenericInterface::PropertyModified;
169 
170  if (!d->propertyMap.contains(key)) {
171  change_type = Solid::GenericInterface::PropertyAdded;
172  }
173 
174  d->propertyMap[key] = value;
175 
176  QMap<QString, int> change;
177  change[key] = change_type;
178 
179  emit d->propertyChanged(change);
180 
181  return true;
182 }

254 QObject *FakeDevice::createDeviceInterface(const Solid::DeviceInterface::Type &type)
255 {
256  // Do not try to cast with a unsupported device interface.
257  if (!queryDeviceInterface(type)) {
258  return 0;
259  }
260 
261  FakeDeviceInterface *iface = 0;
262 
263  switch (type) {
264  case Solid::DeviceInterface::GenericInterface:
265  iface = new FakeGenericInterface(this);
266  break;
267  case Solid::DeviceInterface::Processor:
268  iface = new FakeProcessor(this);
269  break;
270  case Solid::DeviceInterface::Block:
271  iface = new FakeBlock(this);
272  break;
273  case Solid::DeviceInterface::StorageDrive:
274  iface = new FakeStorage(this);
275  break;
276  case Solid::DeviceInterface::OpticalDrive:
277  iface = new FakeCdrom(this);
278  break;
279  case Solid::DeviceInterface::StorageVolume:
280  iface = new FakeVolume(this);
281  break;
282  case Solid::DeviceInterface::OpticalDisc:
283  iface = new FakeOpticalDisc(this);
284  break;
285  case Solid::DeviceInterface::StorageAccess:
286  iface = new FakeStorageAccess(this);
287  break;
288  case Solid::DeviceInterface::Camera:
289  iface = new FakeCamera(this);
290  break;
291  case Solid::DeviceInterface::PortableMediaPlayer:
292  iface = new FakePortableMediaPlayer(this);
293  break;
294  case Solid::DeviceInterface::Battery:
295  iface = new FakeBattery(this);
296  break;
297  case Solid::DeviceInterface::NetworkShare:
298  iface = new FakeNetworkShare(this);
299  break;
300  case Solid::DeviceInterface::Unknown:
301  break;
302  case Solid::DeviceInterface::Last:
303  break;
304  }
305 
306  if (iface) {
307  QDBusConnection::sessionBus().registerObject(d->udi + '/' + Solid::DeviceInterface::typeToString(type), iface,
308  QDBusConnection::ExportNonScriptableSlots);
309  }
310 
311  return iface;
312 }
