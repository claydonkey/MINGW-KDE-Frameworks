137 void Manager::introspect(const QString &path, bool checkOptical)
138 {
139  QDBusMessage call = QDBusMessage::createMethodCall(UD2_DBUS_SERVICE, path,
140  DBUS_INTERFACE_INTROSPECT, "Introspect");
141  QDBusPendingReply<QString> reply = QDBusConnection::systemBus().call(call);
142 
143  if (reply.isValid()) {
144  QDomDocument dom;
145  dom.setContent(reply.value());
146  QDomNodeList nodeList = dom.documentElement().elementsByTagName("node");
147  for (int i = 0; i < nodeList.count(); i++) {
148  QDomElement nodeElem = nodeList.item(i).toElement();
149  if (!nodeElem.isNull() && nodeElem.hasAttribute("name")) {
150  const QString udi = path + "/" + nodeElem.attribute("name");
151 
152  if (checkOptical) {
153  Device device(udi);
154  if (device.mightBeOpticalDisc()) {
155  QDBusConnection::systemBus().connect(UD2_DBUS_SERVICE, udi, DBUS_INTERFACE_PROPS, "PropertiesChanged", this,
156  SLOT(slotMediaChanged(QDBusMessage)));
157  if (!device.isOpticalDisc()) { // skip empty CD disc
158  continue;
159  }
160  }
161  }
162 
163  m_deviceCache.append(udi);
164  }
165  }
166  } else {
167  qWarning() << "Failed enumerating UDisks2 objects:" << reply.error().name() << "\n" << reply.error().message();
168  }
169 }

181 void Manager::slotInterfacesAdded(const QDBusObjectPath &object_path, const VariantMapMap &interfaces_and_properties)
182 {
183  const QString udi = object_path.path();
184 
185  /* Ignore jobs */
186  if (udi.startsWith(UD2_DBUS_PATH_JOBS)) {
187  return;
188  }
189 
190  qDebug() << udi << "has new interfaces:" << interfaces_and_properties.keys();
191 
192  updateBackend(udi);
193 
194  // new device, we don't know it yet
195  if (!m_deviceCache.contains(udi)) {
196  m_deviceCache.append(udi);
197  emit deviceAdded(udi);
198  }
199  // re-emit in case of 2-stage devices like N9 or some Android phones
200  else if (m_deviceCache.contains(udi) && interfaces_and_properties.keys().contains(UD2_DBUS_INTERFACE_FILESYSTEM)) {
201  emit deviceAdded(udi);
202  }
203 }

227 void Manager::slotMediaChanged(const QDBusMessage &msg)
228 {
229  const QVariantMap properties = qdbus_cast<QVariantMap>(msg.arguments().at(1));
230 
231  if (!properties.contains("Size")) { // react only on Size changes
232  return;
233  }
234 
235  const QString udi = msg.path();
236  updateBackend(udi);
237  qulonglong size = properties.value("Size").toULongLong();
238  qDebug() << "MEDIA CHANGED in" << udi << "; size is:" << size;
239 
240  if (!m_deviceCache.contains(udi) && size > 0) { // we don't know the optdisc, got inserted
241  m_deviceCache.append(udi);
242  emit deviceAdded(udi);
243  }
244 
245  if (m_deviceCache.contains(udi) && size == 0) { // we know the optdisc, got removed
246  emit deviceRemoved(udi);
247  m_deviceCache.removeAll(udi);
248  DeviceBackend::destroyBackend(udi);
249  }
250 }

261 void Manager::updateBackend(const QString &udi)
262 {
263  DeviceBackend *backend = DeviceBackend::backendForUDI(udi);
264  if (!backend) {
265  return;
266  }
267 
268  //This doesn't emit "changed" signals. Signals are emitted later by DeviceBackend's slots
269  backend->allProperties();
270 
271  QVariant driveProp = backend->prop("Drive");
272  if (!driveProp.isValid()) {
273  return;
274  }
275 
276  QDBusObjectPath drivePath = qdbus_cast<QDBusObjectPath>(driveProp);
277  DeviceBackend *driveBackend = DeviceBackend::backendForUDI(drivePath.path(), false);
278  if (!driveBackend) {
279  return;
280  }
281 
282  driveBackend->invalidateProperties();
283 }
