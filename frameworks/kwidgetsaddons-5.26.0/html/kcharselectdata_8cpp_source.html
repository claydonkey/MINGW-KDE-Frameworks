92 bool KCharSelectData::openDataFile()
93 {
94  if (!dataFile.isEmpty()) {
95  return true;
96  } else {
97  QFile file(QStandardPaths::locate(QStandardPaths::GenericDataLocation, QStringLiteral("kf5/kcharselect/kcharselect-data")));
98  if (!file.open(QIODevice::ReadOnly)) {
99  return false;
100  }
101  dataFile = file.readAll();
102  file.close();
103  if (dataFile.size() < 40) {
104  dataFile.clear();
105  return false;
106  }
107  const uchar *data = reinterpret_cast<const uchar *>(dataFile.constData());
108  const quint32 offsetBegin = qFromLittleEndian<quint32>(data + 20);
109  const quint32 offsetEnd = qFromLittleEndian<quint32>(data + 24);
110  uint blocks = (offsetEnd - offsetBegin) / 4;
111  if (blocks <= 167) { // maximum possible number of blocks in BMP
112  // no remapping
113  remapType = -1;
114  } else if (blocks == 174) {
115  // remapping introduced in 5.25
116  remapType = 0;
117  } else {
118  // unknown remapping, abort
119  dataFile.clear();
120  return false;
121  }
122  futureIndex = (new RunIndexCreation(this, dataFile))->start();
123  return true;
124  }
125 }

130 quint16 KCharSelectData::mapCodePointToDataBase(uint code) const
131 {
132  if (remapType == 0) {

162 quint32 KCharSelectData::getDetailIndex(uint c) const
163 {
164  const uchar *data = reinterpret_cast<const uchar *>(dataFile.constData());
165  // Convert from little-endian, so that this code works on PPC too.
166  // http://bugs.debian.org/cgi-bin/bugreport.cgi?bug=482286
167  const quint32 offsetBegin = qFromLittleEndian<quint32>(data + 12);
168  const quint32 offsetEnd = qFromLittleEndian<quint32>(data + 16);
169 
170  int min = 0;
171  int mid;
172  int max = ((offsetEnd - offsetBegin) / 27) - 1;
173 
174  quint16 unicode = mapCodePointToDataBase(c);
175  if (unicode == 0xFFFF) {
176  return 0;
177  }
178 
179  static quint16 most_recent_searched;
180  static quint32 most_recent_result;
181 
182  if (unicode == most_recent_searched) {
183  return most_recent_result;
184  }
185 
186  most_recent_searched = unicode;
187 
188  while (max >= min) {
189  mid = (min + max) / 2;
190  const quint16 midUnicode = qFromLittleEndian<quint16>(data + offsetBegin + mid * 27);
191  if (unicode > midUnicode) {
192  min = mid + 1;
193  } else if (unicode < midUnicode) {
194  max = mid - 1;
195  } else {
196  most_recent_result = offsetBegin + mid * 27;
197 
198  return most_recent_result;
199  }
200  }
201 
202  most_recent_result = 0;
203  return 0;
204 }

305 QString KCharSelectData::name(uint c)
306 {
307  if (!openDataFile()) {
308  return QString();
309  }
310 
311  if ((c & 0xFFFE) == 0xFFFE || (c >= 0xFDD0 && c <= 0xFDEF)) {
312  return QCoreApplication::translate("KCharSelectData", "<noncharacter>");
313  } else if ((c >= 0x3400 && c <= 0x4DBF)
314  || (c >= 0x4E00 && c <= 0x9FFF)
315  || (c >= 0x20000 && c <= 0x2F7FF)) {
316  return QStringLiteral("CJK UNIFIED IDEOGRAPH-") + formatCode(c, 4, QString());
317  } else if (c >= 0xAC00 && c <= 0xD7AF) {
318  /* compute hangul syllable name as per UAX #15 */
319  int SIndex = c - SBase;
320  int LIndex, VIndex, TIndex;
321 
322  if (SIndex < 0 || SIndex >= SCount) {
323  return QString();
324  }
325 
326  LIndex = SIndex / NCount;
327  VIndex = (SIndex % NCount) / TCount;
328  TIndex = SIndex % TCount;
329 
330  return QLatin1String("HANGUL SYLLABLE ") + QLatin1String(JAMO_L_TABLE[LIndex])
331  + QLatin1String(JAMO_V_TABLE[VIndex]) + QLatin1String(JAMO_T_TABLE[TIndex]);
332  } else if (c >= 0xD800 && c <= 0xDB7F) {
333  return QCoreApplication::translate("KCharSelectData", "<Non Private Use High Surrogate>");
334  } else if (c >= 0xDB80 && c <= 0xDBFF) {
335  return QCoreApplication::translate("KCharSelectData", "<Private Use High Surrogate>");
336  } else if (c >= 0xDC00 && c <= 0xDFFF) {
337  return QCoreApplication::translate("KCharSelectData", "<Low Surrogate>");
338  } else if ((c >= 0xE000 && c <= 0xF8FF) || c >= 0xF0000) {
339  return QCoreApplication::translate("KCharSelectData", "<Private Use>");
340  } else if ((c >= 0xF900 && c <= 0xFAFF) || (c >= 0x2F800 && c <= 0x2FFFF)) {
341  return QStringLiteral("CJK COMPATIBILITY IDEOGRAPH-") + formatCode(c, 4, QString());
342  }
343  quint16 unicode = mapCodePointToDataBase(c);
344  if (unicode == 0xFFFF) {
345  return QStringLiteral("NON-BMP-CHARACTER-") + formatCode(c, 4, QString());
346  } else {
347  const uchar *data = reinterpret_cast<const uchar *>(dataFile.constData());
348  const quint32 offsetBegin = qFromLittleEndian<quint32>(data + 4);
349  const quint32 offsetEnd = qFromLittleEndian<quint32>(data + 8);
350 
351  int min = 0;
352  int mid;
353  int max = ((offsetEnd - offsetBegin) / 6) - 1;
354  QString s;
355 
356  while (max >= min) {
357  mid = (min + max) / 2;
358  const quint16 midUnicode = qFromLittleEndian<quint16>(data + offsetBegin + mid * 6);
359  if (unicode > midUnicode) {
360  min = mid + 1;
361  } else if (unicode < midUnicode) {
362  max = mid - 1;
363  } else {
364  quint32 offset = qFromLittleEndian<quint32>(data + offsetBegin + mid * 6 + 2);
365  s = QString::fromUtf8(dataFile.constData() + offset + 1);
366  break;
367  }
368  }
369 
370  if (s.isNull()) {
371  return QCoreApplication::translate("KCharSelectData", "<not assigned>");
372  } else {
373  return s;
374  }
375  }
376 }

666  categoryCode--; /* Qt5 changed QChar::Category enum to start from 0 instead of 1
667  See QtBase commit d17c76feee9eece4 */
668  return QChar::Category(categoryCode);
669  }
670  }
671 
672  return QChar::category(c);

693 bool KCharSelectData::isIgnorable(uint c)
694 {
695  /*
696  * According to the Unicode standard, Default Ignorable Code Points
697  * should be ignored unless explicitly supported. For example, U+202E
698  * RIGHT-TO-LEFT-OVERRIDE ir printable according to Qt, but displaying
699  * it gives the undesired effect of all text being turned RTL. We do not
700  * have a way to "explicitly" support it, so we will treat it as
701  * non-printable.
702  *
703  * There is a list of these on
704  * http://unicode.org/Public/UNIDATA/DerivedCoreProperties.txt under the
705  * property Default_Ignorable_Code_Point.
706  */
707 
708  //NOTE: not very nice to hardcode these here; is it worth it to modify
709  // the binary data file to hold them?
710  return c == 0x00AD || c == 0x034F || c == 0x115F || c == 0x1160 ||
711  c == 0x17B4 || c == 0x17B5 || (c >= 0x180B && c <= 0x180D) ||
712  (c >= 0x200B && c <= 0x200F) || (c >= 0x202A && c <= 0x202E) ||
713  (c >= 0x2060 && c <= 0x206F) || c == 0x3164 ||
714  (c >= 0xFE00 && c <= 0xFE0F) || c == 0xFEFF || c == 0xFFA0 ||
715  (c >= 0xFFF0 && c <= 0xFFF8);
716 }

796 QVector<uint> KCharSelectData::find(const QString &needle)
797 {
798  QSet<uint> result;
799 
800  QVector<uint> returnRes;
801  QString simplified = needle.simplified();
802  QStringList searchStrings;
803 
804  QRegularExpression octalExp(QStringLiteral("^\\\\[0-7][0-7\\\\]*$"));
805  QRegularExpressionMatch match = octalExp.match(simplified);
806  if (match.hasMatch()) {
807  // search for C octal escaped UTF-8
808  QByteArray utf8;
809  int byte = -1;
810  for (int i = 0; i <= simplified.length(); ++i) {
811  int c = simplified.at(i).unicode();
812  if (c >= '0' && c <= '7') {
813  byte = 8 * byte + c - '0';
814  } else if (byte == -1) {
815  byte = 0;
816  } else if (byte >= 0x00 && byte <= 0xFF) {
817  utf8.append((char) byte);
818  byte = 0;
819  }
820  }
821  simplified = QString::fromUtf8(utf8);
822  }
823 
824  if (simplified.length() <= 2) {
825  QVector<uint> ucs4 = simplified.toUcs4();
826  if (ucs4.size() == 1) {
827  // search for hex representation of the character
828  searchStrings = QStringList(formatCode(ucs4.at(0)));
829  }
830  } else {
831  searchStrings = splitString(simplified);
832  }
833 
834  if (searchStrings.count() == 0) {
835  return returnRes;
836  }
837 
838  QRegularExpression hexExp(QStringLiteral("^(|u\\+|U\\+|0x|0X)([A-Fa-f0-9]{4,5})$"));
839  foreach (const QString &s, searchStrings) {
840  QRegularExpressionMatch match = hexExp.match(s);
841  if (match.hasMatch()) {
842  returnRes.append(match.captured(2).toInt(0, 16));
843  // search for "1234" instead of "0x1234"
844  if (s.length() == 6 || s.length() == 7) {
845  searchStrings[searchStrings.indexOf(s)] = match.captured(2);
846  }
847  }
848  // try to parse string as decimal number
849  bool ok;
850  int unicode = s.toInt(&ok);
851  if (ok && unicode >= 0 && unicode <= QChar::LastValidCodePoint) {
852  returnRes.append(unicode);
853  }
854  }
855 
856  bool firstSubString = true;
857  foreach (const QString &s, searchStrings) {
858  QSet<uint> partResult = getMatchingChars(s.toLower());
859  if (firstSubString) {
860  result = partResult;
861  firstSubString = false;
862  } else {
863  result = result.intersect(partResult);
864  }
865  }
866 
867  // remove results found by matching the code point to prevent duplicate results
868  // while letting these characters stay at the beginning
869  foreach (uint c, returnRes) {
870  result.remove(c);
871  }
872 
873  QVector<uint> sortedResult;
874  sortedResult.reserve(result.count());
875  QSet<uint>::const_iterator it = result.begin();
876  const QSet<uint>::const_iterator end = result.end();
877  for ( ; it != end ; ++it ) {
878  sortedResult.append(*it);
879  }
880  qSort(sortedResult);
881 
882  returnRes += sortedResult;
883  return returnRes;
884 }

906 QStringList KCharSelectData::splitString(const QString &s)
907 {
908  QStringList result;
909  int start = 0;
910  int end = 0;
911  int length = s.length();
912  while (end < length) {
913  while (end < length && (s[end].isLetterOrNumber() || s[end] == QLatin1Char('+'))) {
914  end++;
915  }
916  if (start != end) {
917  result.append(s.mid(start, end - start));
918  }
919  start = end;
920  while (end < length && !(s[end].isLetterOrNumber() || s[end] == QLatin1Char('+'))) {
921  end++;
922  start++;
923  }
924  }
925  return result;
