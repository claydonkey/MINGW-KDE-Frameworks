192 void KLed::updateAccessibleName()
193 {
194 #ifndef QT_NO_ACCESSIBILITY
195  QString onName = tr("LED on", "Accessible name of a Led whose state is on");
196  QString offName = tr("LED off", "Accessible name of a Led whose state is off");
197  QString lastName = accessibleName();
198 
199  if (lastName.isEmpty() || lastName == onName || lastName == offName) {
200  //Accessible name has not been manually set.
201 
202  setAccessibleName(d->state == On ? onName : offName);
203  }
204 #endif
205 }

214 void KLed::paintEvent(QPaintEvent *)
215 {
216  if (!d->cachedPixmap[d->state].isNull()) {
217  QPainter painter(this);
218  painter.drawPixmap(1, 1, d->cachedPixmap[d->state]);
219  return;
220  }
221 
222  QSize size(width() - 2, height() - 2);
223  if (d->shape == Circular) {
224  // Make sure the LED is round
225  const int dim = qMin(width(), height()) - 2;
226  size = QSize(dim, dim);
227  }
228  QPointF center(size.width() / 2.0, size.height() / 2.0);
229  const int smallestSize = qMin(size.width(), size.height());
230  QPainter painter;
231 
232  QImage image(size, QImage::Format_ARGB32_Premultiplied);
233  image.fill(0);
234 
235  QRadialGradient fillGradient(center, smallestSize / 2.0, QPointF(center.x(), size.height() / 3.0));
236  const QColor fillColor = d->state != Off ? d->color : d->color.dark(d->darkFactor);
237  fillGradient.setColorAt(0.0, fillColor.light(250));
238  fillGradient.setColorAt(0.5, fillColor.light(130));
239  fillGradient.setColorAt(1.0, fillColor);
240 
241  QConicalGradient borderGradient(center, d->look == Sunken ? 90 : -90);
242  QColor borderColor = palette().color(QPalette::Dark);
243  if (d->state == On) {
244  QColor glowOverlay = fillColor;
245  glowOverlay.setAlpha(80);
246 
247  // This isn't the fastest way, but should be "fast enough".
248  // It's also the only safe way to use QPainter::CompositionMode
249  QImage img(1, 1, QImage::Format_ARGB32_Premultiplied);
250  QPainter p(&img);
251  QColor start = borderColor;
252  start.setAlpha(255); // opaque
253  p.fillRect(0, 0, 1, 1, start);
254  p.setCompositionMode(QPainter::CompositionMode_SourceOver);
255  p.fillRect(0, 0, 1, 1, glowOverlay);
256  p.end();
257 
258  borderColor = img.pixel(0, 0);
259  }
260  borderGradient.setColorAt(0.2, borderColor);
261  borderGradient.setColorAt(0.5, palette().color(QPalette::Light));
262  borderGradient.setColorAt(0.8, borderColor);
263 
264  painter.begin(&image);
265  painter.setRenderHint(QPainter::Antialiasing);
266  painter.setBrush(d->look == Flat ? QBrush(fillColor) : QBrush(fillGradient));
267  const QBrush penBrush = (d->look == Flat) ? QBrush(borderColor) : QBrush(borderGradient);
268  const qreal penWidth = smallestSize / 8.0;
269  painter.setPen(QPen(penBrush, penWidth));
270  QRectF r(penWidth / 2.0, penWidth / 2.0, size.width() - penWidth, size.height() - penWidth);
271  if (d->shape == Rectangular) {
272  painter.drawRect(r);
273  } else {
274  painter.drawEllipse(r);
275  }
276  painter.end();
277 
278  d->cachedPixmap[d->state] = QPixmap::fromImage(image);
279  painter.begin(this);
280  painter.drawPixmap(1, 1, d->cachedPixmap[d->state]);
281  painter.end();
282 }
