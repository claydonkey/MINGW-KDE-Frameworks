90 void KNewPasswordWidget::KNewPasswordWidgetPrivate::_k_textChanged()
91 {
92  const QString password = ui.linePassword->text();
93  const QString verification = ui.lineVerifyPassword->text();
94  const bool match = (password == verification);
95  const bool partialMatch = password.startsWith(verification);
96  const int minPasswordLength = q->minimumPasswordLength();
97 
98  QPalette palette = q->palette();
99  palette.setColor(QPalette::Active, QPalette::Base, (match || partialMatch) ? defaultBackgroundColor : backgroundWarningColor);
100  ui.lineVerifyPassword->setPalette(palette);
101 
102  // Password strength calculator
103  int pwstrength = (20 * ui.linePassword->text().length() + 80 * effectivePasswordLength(ui.linePassword->text())) / qMax(reasonablePasswordLength, 2);
104  ui.strengthBar->setValue(qBound(0, pwstrength, 100));
105 
106  // update the current password status
107  if (match || ui.lineVerifyPassword->isHidden()) {
108  if (!q->allowEmptyPasswords() && ui.linePassword->text().isEmpty()) {
109  updatePasswordStatus(EmptyPasswordNotAllowed);
110  } else if (ui.linePassword->text().length() < minPasswordLength) {
111  updatePasswordStatus(PasswordTooShort);
112  } else if (ui.strengthBar && ui.strengthBar->value() < passwordStrengthWarningLevel) {
113  updatePasswordStatus(WeakPassword);
114  } else {
115  updatePasswordStatus(StrongPassword);
116  }
117  } else {
118  updatePasswordStatus(PasswordNotVerified);
119  }
120 }

144 int KNewPasswordWidget::KNewPasswordWidgetPrivate::effectivePasswordLength(const QString &password)
145 {
146  enum Category {
147  Digit,
148  Upper,
149  Vowel,
150  Consonant,
151  Special
152  };
153 
154  Category previousCategory = Vowel;
155  QString vowels(QStringLiteral("aeiou"));
156  int count = 0;
157 
158  for (int i = 0; i < password.length(); ++i) {
159  QChar currentChar = password.at(i);
160  if (!password.leftRef(i).contains(currentChar)) {
161  Category currentCategory;
162  switch (currentChar.category()) {
163  case QChar::Letter_Uppercase:
164  currentCategory = Upper;
165  break;
166  case QChar::Letter_Lowercase:
167  if (vowels.contains(currentChar)) {
168  currentCategory = Vowel;
169  } else {
170  currentCategory = Consonant;
171  }
172  break;
173  case QChar::Number_DecimalDigit:
174  currentCategory = Digit;
175  break;
176  default:
177  currentCategory = Special;
178  break;
179  }
180  switch (currentCategory) {
181  case Vowel:
182  if (previousCategory != Consonant) {
183  ++count;
184  }
185  break;
186  case Consonant:
187  if (previousCategory != Vowel) {
188  ++count;
189  }
190  break;
191  default:
192  if (previousCategory != currentCategory) {
193  ++count;
194  }
195  break;
196  }
197  previousCategory = currentCategory;
198  }
199  }
200  return count;
201 }
