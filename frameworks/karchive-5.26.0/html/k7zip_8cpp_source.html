587 quint64 K7Zip::K7ZipPrivate::readNumber()
588 {
589  if (!buffer || (quint64)(pos + 8) > end) {
590  return 0;
591  }
592 
593  unsigned char firstByte = buffer[pos++];
594  unsigned char mask = 0x80;
595  quint64 value = 0;
596  for (int i = 0; i < 8; i++) {
597  if ((firstByte & mask) == 0) {
598  quint64 highPart = firstByte & (mask - 1);
599  value += (highPart << (i * 8));
600  return value;
601  }
602  value |= ((unsigned char)buffer[pos++] << (8 * i));
603  mask >>= 1;
604  }
